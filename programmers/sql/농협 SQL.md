```sql
SELECT BANK_ID FROM BANK_TABLE
WHERE (DATE_FORMAT(CREATED_ID, '%h-%m-%s') < 09-00-00 OR DATE_FORMAT(CREATED_ID, '%h-%m-%s') > 17-00-00) AND AMOUNT >= 1000000
GROUP BY BANK_ID
ORDER BY BANK_ID
```
